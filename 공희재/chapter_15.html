Chapter 15 - let, const 키워드와 블록 레벨 스코프#

1. var 키워드의 문제점#

var 키워드로 선언한 변수는 다음과 같은 특징이 있으며, 주의를 기울이지 않으면 심각한 문제를 발생시킬 수 있다.

1-1. 변수 중복 선언을 허용#

var 키워드로는 아래처럼 중복 선언을 할 수 있다.
var 키워드를 사용할 때 이미 선언된 변수를 모르고 같은 이름으로 중복 선언을 할 경우
의도치 않게 먼저 선언한 변수 값을 변경해 버리는 부작용이 발생할 수 있다.

var x = 1;
var y = 1;

// 같은 스코프에서 중복 선언을 할 수 있다.
var x = 100;  // 이렇게 초기화문이 있는 변수 선언문은 마치 var 키워드가 없는 것처럼 동작한다.
var y;  // 이렇게 초기화문이 없으면 그냥 무시됨.

console.log(x)  // 100
console.log(y)  // 1

1-2. 함수 레벨 스코프#

var로 선언한 변수는 오직 함수의 코드 블록만을 지역 스코프로 인정한다.
아래처럼 코드 블록 안에서 새로 선언한 변수가, 함수 밖에서 var로 선언한 변수와 이름이 같다면,
그 변수는 새로 선언한 변수가 아닌 중복 선언한 전역 변수가 될 뿐이다.

var x = 1;

if (true) {
  var x = 10;  // 여기서 x는 이미 위에 선언한 전역 변수 x가 있으므로, 중복 선언한
}

console.log(x);  // 10
